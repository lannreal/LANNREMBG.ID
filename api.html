# Remove background dengan cURL
curl -X POST http://localhost:3000/api/remove-bg \
  -H "X-API-Key: YOUR_API_KEY" \
  -F "image=@/path/to/gambar.jpg" \
  -F "model=u2netp" \
  --output hasil.png
const formData = new FormData();
formData.append('image', fileInput.files[0]);
formData.append('model', 'u2netp');

const res = await fetch('http://localhost:3000/api/remove-bg', {
  method: 'POST',
  headers: { 'X-API-Key': 'YOUR_API_KEY' },
  body: formData
});

const data = await res.json();
console.log(data.resultUrl); // URL gambar hasil
import requests

with open('gambar.jpg', 'rb') as f:
    res = requests.post(
        'http://localhost:3000/api/remove-bg',
        headers={'X-API-Key': 'YOUR_API_KEY'},
        files={'image': f},
        data={'model': 'u2netp'}
    )

data = res.json()
print(data['resultUrl'])  # URL gambar hasil
$curl = curl_init();
curl_setopt_array($curl, [
  CURLOPT_URL            => 'http://localhost:3000/api/remove-bg',
  CURLOPT_POST           => true,
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_HTTPHEADER     => ['X-API-Key: YOUR_API_KEY'],
  CURLOPT_POSTFIELDS     => [
    'image' => new CURLFile('gambar.jpg'),
    'model' => 'u2netp'
  ]
]);
$response = json_decode(curl_exec($curl), true);
echo $response['resultUrl'];
